49 /**
50  * This class is a dense univariate polynomial.
51  *
52  * This polynomial has a size of at least one - degree 0. The zero polynomial
53  * has a_0 = 0. There is no null polynomial. size == degree + 1
54  *
55  * How to handle division?
56  * operator/ returns the quotient of two polynomials discarding the remainder.
57  * operator% returns the remainder of two polynomials discarding the quotient.
58  * The divmod functions computes both the quotient and the remainder.
59  * N. B. I could add the remquo function?
60  * *
61  * Largest coefficient:
62  * This class does not enforce the coefficient of largest power to be nonzero.
63  * The user concerned about this should use the deflate method with a
64  * tolerance for the minimim magnitude coefficient.
65  * This library provides another deflate method that divides out a polynomial
66  * that is expected to factor the polynomial within a tolerance on the
67  * minimim magnitude coefficient. If the remainder is non-zero then an
68  * exception is thrown.
69  *
70  * Mixed-type evaluation:
71  * It is common to have integer coefficient polynomials that you wish to
72  * evaluate at floating opint or complex values, or real-valued coefficient
73  * polynomials evaluate at complex values. For this reason we need evaluation
74  * function templates in this library.
75  *
76  * It would be promote_t<complex::value_type> for complex.
77  *
78  */

207  /**
208  * Use Lagrange interpolation to construct a polynomial passing through
209  * the data points. The degree will be one less than the number of points.
210  */
211  template<typename InIter,
212  typename = std::_RequireInputIter<InIter>>
213  Polynomial(const InIter& xbegin, const InIter& xend,
214  const InIter& ybegin)
215  : m_coeff()
216  {
217  std::vector<Polynomial<value_type>> numer;
218  std::vector<Polynomial<value_type>> denom;
219  for (auto xi = xbegin; xi != xend; ++xi)
220  {
221  for (auto xj = xi + 1; xj != xend; ++xj)
222  denom.push_back(value_type(*xj) - value_type(*xi));
223  numer.push_back({-value_type(*xi), value_type{1}});
224  }
225  this->m_set_scale();
226  }

248  /**
249  * Evaluate the polynomial at the input point.
250  */
251  value_type
252  operator()(value_type x) const
253  {
254  if (this->degree() > 0)
255  {
256  if (std::abs(x) <= real_type{1})
257  {
258  value_type poly(this->coefficient(this->degree()));
259  for (int i = this->degree() - 1; i >= 0; --i)
260  poly = poly * x + this->coefficient(i);
261  return poly;
262  }
263  else
264  {
265  const auto rx = real_type{1} / x;
266  value_type poly(this->coefficient(0));
267  for (std::size_t i = 1ull; i <= this->degree(); ++i)
268  poly = poly * rx + this->coefficient(i);
269  for (std::size_t i = 1ull; i <= this->degree(); ++i)
270  poly *= x;
271  return poly;
272  }
273  }
274  else
275  return this->coefficient(0);
276  }

278  /**
279  * Evaluate the polynomial at the input point.
280  */
281  template<typename Up>
282  auto
283  operator()(Up x) const
284  -> decltype(value_type{} * Up{})
285  {
286  if (this->degree() > 0)
287  {
288  if (std::abs(x) <= real_type{1})
289  {
290  auto poly(Up{1} * this->coefficient(this->degree()));
291  for (int i = this->degree() - 1; i >= 0; --i)
292  poly = poly * x + this->coefficient(i);
293  return poly;
294  }
295  else
296  {
297  const auto rx = real_type{1} / x;
298  auto poly(Up{1} * this->coefficient(0));
299  for (std::size_t i = 1ull; i <= this->degree(); ++i)
300  poly = poly * rx + this->coefficient(i);
301  for (std::size_t i = 1ull; i <= this->degree(); ++i)
302  poly *= x;
303  return poly;
304  }
305  }
306  else
307  return Up{1} * this->coefficient(0);
308  }

312  * exploits the fact that the polynomial coefficients are all real.

314  * The algorithm is discussed in detail in:

408  /**
409  * Return the derivative polynomial.
410  */
411  Polynomial
412  derivative() const
413  {
414  Polynomial res(value_type{},
415  this->degree() > 0UL ? this->degree() - 1 : 0UL);
416  for (size_type n = this->degree(), i = 1; i <= n; ++i)
417  res.m_coeff[i - 1] = i * this->m_coeff[i];
418  return res;
419  }
420 
421  /**
422  * Return the derivative of the polynomial at the given point.
423  */
424  template<typename Up>
425  decltype(Up{} * value_type{})
426  derivative(Up c) const
427  {
428  using res_t = decltype(Up{} * value_type{});
429  const int n = this->degree();
430  res_t res = real_type(n) * this->m_coeff[n];
431  for (int i = n - 1; i > 0; --i)
432  res = c * res + real_type(i) * this->m_coeff[i];
433  return res;
434  }

436  /**
437  * Return the integral polynomial with given integration constant.
438  */
439  Polynomial
440  integral(value_type c = value_type{}) const
441  {
442  Polynomial res(value_type{}, this->degree() + 1);
443  res.m_coeff[0] = c;
444  for (size_type n = this->degree(), i = 0; i <= n; ++i)
445  res.m_coeff[i + 1] = this->m_coeff[i] / value_type(i + 1);
446  return res;
447  }

449  /**
450  * Return the integral of the polynomial with given integration limits.
451  */
452  template<typename Up>
453  decltype(Up{} * value_type{})
454  integral(Up a, Up b) const
455  {
456  using res_t = decltype(Up{} * value_type{});
457  const int n = this->degree();
458  const auto coeff = this->m_coeff[n] / real_type(n + 1);
459  res_t resa = coeff * a;
460  res_t resb = coeff * b;
461  for (int i = n - 1; i >= 0; --i)
462  {
463  const auto coeff = this->m_coeff[i] / real_type(i + 1);
464  resa += coeff;
465  resa *= a;
466  resb += coeff;
467  resb *= b;
468  }
469  return resb - resa;
470  }

645  /**
646  * Shift the polynomial using the Horner scheme.
647  * Given our polynomial
648  * @f[
649  * P(x) = a_0 + a_1 x + a_2 x^2 + ...
650  * @f]
651  * Obtain a new polynomial
652  * @f[
653  * Q(z) = P(x + s) = a_0 + a_1 (x + s) + a_2 (x + s)^2 + ... = b_0 + b_1 x + b_2 x^2
654  * @f]
655  */
656  void
657  shift(value_type shift)
658  {
659  if (shift == value_type{})
660  return;
661  const int n = this->degree();
662  for (int j = 1; j <= n; ++j)
663  for (int i = 1; i <= n - j + 1; ++i)
664  this->m_coeff[n - i] += shift * this->m_coeff[n - i + 1];
665  }

843  Polynomial&
844  deflate(const Polynomial<value_type>& poly,
845  real_type max_abs_coef)
846  {
847  Polynomial<value_type> quo, rem;
848  divmod(*this, poly, quo, rem);
849

851  size_type n = rem.degree();
852  for (size_type i = rem.degree(); i > 0; --i)
853  if (std::abs(rem[i]) < max_abs_coef)
854  --n;
855  else
856  break;

859  *this = quo.deflate(max_abs_coef);
860  else
861  throw std::runtime_error("deflate: ");
