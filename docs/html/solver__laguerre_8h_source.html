43  /**
44  * A solver for complex-coefficient polynomials due to Laguerre.
45  */
46  template<typename Real>
47  class LaguerreSolver
48  {
49  public:
50 
51  LaguerreSolver(Polynomial<std::complex<Real>>& P)
52  : m_poly(P), m_num_iters{0}
53  { }
54 
55  std::vector<std::complex<Real>> solve();

57  std::complex<Real>
58  step()
59  {
60  using cmplx = std::complex<Real>;
61 
62  const auto z0 = this->m_root_laguerre();
63 
64  Polynomial<cmplx> zpoly({-z0, cmplx{1}});
65  this->m_poly /= zpoly;
66 
67  return z0;
68  }
69 
70  int
71  num_iters() const

74  int

81 
82  const Polynomial<std::complex<Real>>&
83  polynomial() const

86  LaguerreSolver&

88  {
89  this->m_steps_per_frac = num;
90  return *this;
91  }
92 
93  private:
94 
95  // Estimated fractional roundoff error.
96  static constexpr Real s_eps = std::numeric_limits<Real>::epsilon();
97 
98  // Number of fractional values.
99  static constexpr int s_num_fracs = 8;
100  // Fractions used to break a limit cycle (in a heap).
101  static constexpr Real
102  s_frac[s_num_fracs + 1]
103  {0.0, 0.5, 0.25, 0.75, 0.125, 0.375, 0.625, 0.875, 1.0};
104 
105  // Number of steps taken before trying a new fraction.
106  int m_steps_per_frac = 10;
107 
108  int
109  m_max_iter()

115 
116  int m_num_iters = 0;
117  };
