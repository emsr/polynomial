39  /**
40  * I think this is trying to factor out a quadratic
41  * from a complex-coefficient polynomial.
42  */
43  template<typename Tp>
44  Polynomial<std::complex<Tp>>
45  QuadraticSolver<Tp>::m_root_quadratic()
46  {
47  using Cmplx = std::complex<Tp>;
48  using Poly = Polynomial<Cmplx>;
49 
50  if (this->m_poly.degree() <= 2)
51  return this->m_poly;
52 
53  this->m_num_iters = 0;
54 
55  Cmplx c, b;
56  Poly q, qq, rem;
57  for (int iter = 0; iter < this->m_max_iter; ++iter)
58  {
59  ++this->m_num_iters;
60 
61  Poly d({c, b, Cmplx{1}});
62 
63  // First division: r, s.
64  divmod(this->m_poly, d, q, rem);
65  const auto s = rem[0];
66  const auto r = rem[1];
67 
68  // Second division: partial r, s with respect to c.
69  divmod(q, d, qq, rem);
70  const auto sc = -rem[0];
71  const auto rc = -rem[1];
72  const auto sb = -c * rc;
73  const auto rb = -b * rc + sc;
74 
75  // Solve 2x2 equation.
76  const auto dv = Tp{1} / (sb * rc - sc * rb);
77  const auto delb = ( r * sc - s * rc) * dv;
78  b += delb;
79  const auto delc = (-r * sb + s * rb) * dv;
80  c += delc;
81  if ((std::abs(delb) <= this->m_eps * std::abs(b)
82  || std::abs(b) < s_tiny)
83  && (std::abs(delc) <= this->m_eps * std::abs(c)
84  || std::abs(c) < s_tiny))
85  return Poly({c, b, Cmplx{1}});
86  }
87  throw std::runtime_error("m_root_quadratic: Maximum number of iterations exceeded");
88  }
