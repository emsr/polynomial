209  /**
210  * Read a polynomial from a stream.
211  * The input format can be a plain scalar (zero degree polynomial)
212  * or a parenthesized comma-delimited list of coefficients.
213  */
214  template<typename CharT, typename Traits, typename Tp>
215  std::basic_istream<CharT, Traits>&
216  operator>>(std::basic_istream<CharT, Traits>& is, RationalPolynomial<Tp>& poly)
217  {
218  Polynomial<Tp> numer, denom;
219  is >> numer;
220  if (!is.fail())
221  {
222  CharT ch;
223  is >> ch;
224  if (ch != '/')
225  is.setstate(std::ios_base::failbit);
226  else
227  {
228  is >> denom;
229  if (!is.fail())
230  {
231  poly.numer() = numer;
232  poly.denom() = denom;
233  }
234  }
235  }
236  return is;
237  }
