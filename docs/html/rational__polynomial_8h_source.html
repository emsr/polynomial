219  /**
220  * Read a polynomial from a stream.
221  * The input format can be a plain scalar (zero degree polynomial)
222  * or a parenthesized comma-delimited list of coefficients.
223  */
224  template<typename CharT, typename Traits, typename _Tp>
225  std::basic_istream<CharT, Traits>&
226  operator>>(std::basic_istream<CharT, Traits>& __is, _RationalPolynomial<_Tp>& __poly)
227  {
228  _Polynomial<_Tp> __numer, __denom;
229  __is >> __numer;
230  if (!__is.fail())
231  {
232  CharT __ch;
233  __is >> __ch;
234  if (__ch != '/')
235  __is.setstate(std::ios_base::failbit);
236  else
237  {
238  __is >> __denom;
239  if (!__is.fail())
240  {
241  __poly.numer() = __numer;
242  __poly.denom() = __denom;
243  }
244  }
245  }
246  return __is;
247  }
