48  * Perform compile-time evaluation of a constant zero-order polynomial.
49  */
50 template<typename _ArgT, typename _Coef0>
51  constexpr std::conditional_t<std::is_integral<_ArgT>::value, double, _ArgT>
52  horner(_ArgT __x, _Coef0 __c0)

59 /**
60  * Perform compile-time evaluation of a constant polynomial.
61  * The polynomial coefficients are lowest-order first.
62  */
63 template<typename _ArgT, typename _Coef0, typename... _Coef>
