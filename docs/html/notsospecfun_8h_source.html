282  // Reperiodized trigonometric functions...
283  // fun_pi(x) = fun(pi x);
284 
285  // This is really just another angle unit.
286  // When we get units, this and deg, grad, rad would all get overloads.
287  // We shouldn't need decorated functions.
288  // OTOH, machines have these - there are built-ins and traditions...
289  //
290  // I want to have minimum regret wrt future units.
291  // The party would start with the inverses...
292  // You may want only rad units to have a non-explicit ctor/assingments
293  // from floating point numbers (we can't oload on return type).
294  // Or rather other units would feed floating point numbers through rad.
295 
296  // We need an opaque typedef for reperiodized angles.
297  // We don't need to introduce new opportunities for errors.
298  // The type will have an implicit conversion to floating point radians
299  // so the output of reperiodized inverse functions can go into the
300  // pre-existing trigonometric functions as hoped.
301  // The new reperiodized trigonometric functions bear the burden
302  // of providing overloads for reperiod_t arguments
303  // (so we don't get sin(pi^2 x)).
304  template<typename Tp>

307  Tp value;
308  constexpr operator Tp()

449  /**
450  * Give complex an fma.
451  */
452  template<typename Tp>
453  inline std::complex<Tp>
454  fma(const std::complex<Tp>& a, const std::complex<Tp>& z,
455  const std::complex<Tp>& b)
456  {
457  const auto [ar, ai] = reinterpret_cast<const Tp(&)[2]>(a);
458  const auto [zr, zi] = reinterpret_cast<const Tp(&)[2]>(z);
459  const auto [br, bi] = reinterpret_cast<const Tp(&)[2]>(b);
460  const auto wr = std::fma(ar, ai, -std::fma(ai, zi, -br));
461  const auto wi = std::fma(ar, zi, std::fma(ai, zr, bi));
462  return {wr, wi};
463  }
