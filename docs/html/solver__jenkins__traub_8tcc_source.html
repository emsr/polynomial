42 /**
43  * Constructor from input polynomial.
44  */
45 template<typename Real>
46  JenkinsTraubSolver<Real>::
47  JenkinsTraubSolver(const std::vector<Real>& op)
48  : m_P(op)
49  {
50  if (this->m_P.size() == 0)
51  throw std::domain_error("Polynomial degree must be at least 1.");
52 
53  // Algorithm fails if the leading coefficient is zero.
54  // We could erase leading-order zero coefficients.
55  if (this->m_P[0] == Real{0})
56  throw std::domain_error("Leading coefficient must be nonzero.");
57 
58  const auto degree = this->m_P.size() - 1;
59  this->m_order = degree;
60  this->m_P_quot.resize(degree + 1);
61  this->m_H.resize(degree + 1);
62  this->m_H_quot.resize(degree + 1);
63  this->m_H_save.resize(degree + 1);
64  }

66 /**
67  *
68  */
69 template<typename Real>
70  std::vector<Solution<Real>>
71  JenkinsTraubSolver<Real>::solve()
72  {
73  // Initialization of constants for shift rotation.
74  auto xx = 1 / Real{1.4142'13562'37309'50488'01688'72420'96980'78569e+0L};
75  auto yy = -xx;
76  const auto cosr = std::cos(s_rotation);
77  const auto sinr = std::sin(s_rotation);
78 
79  std::vector<Solution<Real>> zero;
80  zero.reserve(this->m_P.size());
81 
82  // Remove the zeros at the origin, if any.
83  while (this->m_P[this->m_order] == Real{0})
84  {
85  zero.push_back(Real{0});
86  --this->m_order;
87  }
88  if (this->m_order < 1)
89  return zero;
90 
91  std::vector<Real> pt(this->m_order + 1);
92  std::vector<Real> _H_temp(this->m_order + 1);
93 
94  while (true)
95  {
96  // Start the algorithm for one zero.
97  this->m_num_iters = 0;
98  if (this->m_order == 1)
99  {
100  zero.push_back(-this->m_P[1] / this->m_P[0]);
101  --this->m_order;
102  return zero;
103  }
104  // Calculate the final zero or pair of zeros.
105  if (this->m_order == 2)
106  {
107  Solution<Real> z_small, z_large;
108  this->quadratic(this->m_P[0], this->m_P[1], this->m_P[2],
109  z_small, z_large);
110  if (z_small.index() != 0)
111  {
112  zero.push_back(z_small);
113  --this->m_order;
114  }
115  if (z_large.index() != 0)
116  {
117  zero.push_back(z_large);
118  --this->m_order;
119  }
120  return zero;
121  }
122 
123  // Find largest and smallest moduli of coefficients.
124  auto a_max = Real{0};
125  auto a_min = s_huge;
126  for (int i = 0; i <= this->m_order; ++i)
127  {
128  const auto x = std::abs(this->m_P[i]);
129  if (x > a_max)
130  a_max = x;
131  if (x != Real{0} && x < a_min)
132  a_min = x;
133  }
134  // Scale if there are large or very tiny coefficients.
135  // Computes a scale factor to multiply the coefficients
136  // of the polynomial. The scaling is done to avoid overflow
137  // and to avoid undetected underflow interfering
138  // with the convergence criterion.
139  // The factor is a power of the base.
140  auto scale = s_low / a_min;
141  bool rescale = true;
142  if (scale > Real{1} && s_huge / scale < a_max)
143  rescale = false;
144  if (scale <= Real{1} && a_max < Real{10})
145  rescale = false;
146 
147  if (rescale)
148  {
149  // Scale polynomial.
150  if (scale == Real{0})
151  scale = s_tiny;
152  const auto l = std::ilogb(scale);
153  const auto factor = std::pow(s_base, l);
154  if (factor != Real{1})
155  for (int i = 0; i <= this->m_order; ++i)
156  this->m_P[i] *= factor;
157  }
158 
159  // Compute lower bound on moduli of roots.
160  for (int i = 0; i <= this->m_order; ++i)
161  pt[i] = std::abs(this->m_P[i]);
162  pt[this->m_order] = -pt[this->m_order];
163  // Compute upper estimate of bound.
164  auto x = std::exp((std::log(-pt[this->m_order])
165  - std::log(pt[0])) / Real(this->m_order));
166  // If Newton step at the origin is better, use it.
167  if (pt[this->m_order - 1] != Real{0})
168  {
169  const auto xm = -pt[this->m_order] / pt[this->m_order - 1];
170  if (xm < x)
171  x = xm;
172  }
173  // Chop the interval (0,x) until ff <= 0.
174  while (true)
175  {
176  auto xm = x * Real{0.1L};
177  auto ff = pt[0];
178  for (int i = 1; i <= this->m_order; ++i)
179  ff = ff * xm + pt[i];
180  if (ff <= Real{0})
181  break;
182  x = xm;
183  }
184  // Do Newton interation until x converges to two decimal places.
185  auto dx = x;
186  while (std::abs(dx / x) > this->m_min_log_deriv)
187  {
188  auto ff = pt[0];
189  auto df = ff;
190  for (int i = 1; i < this->m_order; ++i)
191  {
192  ff = ff * x + pt[i];
193  df = df * x + ff;
194  }
195  ff = ff * x + pt[this->m_order];
196  dx = ff / df;
197  x -= dx;
198  ++this->m_num_iters;
199  }
200  const auto bound = x;
201  // Compute the derivative as the initial _H polynomial
202  // and do 5 steps with no shift.
203  const auto nm1 = this->m_order - 1;
204  for (int i = 1; i < this->m_order; ++i)
205  this->m_H[i] = Real(this->m_order - i) * this->m_P[i]
206  / Real(this->m_order);
207  this->m_H[0] = this->m_P[0];
208  const auto aa = this->m_P[this->m_order];
209  const auto bb = this->m_P[this->m_order - 1];
210  this->m_zerok = (this->m_H[this->m_order - 1] == Real{0});
211  for(int jj = 0; jj < 5; ++jj)
212  {
213  ++this->m_num_iters;
214  auto cc = this->m_H[this->m_order - 1];
215  if (!this->m_zerok)
216  {
217  // Use a scaled form of recurrence if value of H at 0
218  // is nonzero.
219  const auto t = -aa / cc;
220  for (int i = 0; i < nm1; ++i)
221  {
222  const auto j = this->m_order - i - 1;
223  this->m_H[j] = t * this->m_H[j - 1] + this->m_P[j];
224  }
225  this->m_H[0] = this->m_P[0];
226  this->m_zerok = (std::abs(this->m_H[this->m_order - 1])
227  <= Real{10} * s_eps * std::abs(bb));
228  }
229  else
230  {
231  // Use unscaled form of recurrence.
232  for (int i = 0; i < nm1; ++i)
233  {
234  const auto j = this->m_order - i - 1;
235  this->m_H[j] = this->m_H[j - 1];
236  }
237  this->m_H[0] = Real{0};
238  this->m_zerok = (this->m_H[this->m_order - 1] == Real{0});
239  }
240  }
241  // Save H for restarts with new shifts.
242  _H_temp = this->m_H;
243 
244  // Loop to select the quadratic corresponding to each new shift.
245  for (int count = 0; count < 20; ++count)
246  {
247  /* Quadratic corresponds to a Real shift to a
248  * non-real point and its complex conjugate. The point
249  * has modulus bound and amplitude rotated by 94 degrees
250  * from the previous shift.
251  */
252  const auto xxx = cosr * xx - sinr * yy;
253  yy = sinr * xx + cosr * yy;
254  auto xx = xxx;
255  this->m_sr = bound * xx;
256  this->m_si = bound * yy;
257  this->m_u = -Real{2} * this->m_sr;
258  this->m_v = bound;
259  auto num_zeros = this->fxshfr(20 * (count + 1));
260  bool cycle = false;
261  if (num_zeros != 0)
262  {
263  /* The second stage jumps directly to one of the third
264  * stage iterations and returns here if successful.
265  * Deflate the polynomial, store the zero or zeros
266  * and return to the main algorithm.
267  */
268  zero.push_back(this->m_z_small);
269  this->m_order -= num_zeros;
270  this->m_P = this->m_P_quot;
271  if (num_zeros != 1)
272  zero.push_back(this->m_z_large);
273  cycle = true;
274  break;
275  }
276  if (cycle)
277  continue;
278 
279  // If the iteration is unsuccessful another quadratic
280  // is chosen after restoring H.
281  this->m_H = _H_temp;
282  }
283  }
284  }

287 /**
288  * Computes up to L2 fixed shift H-polynomials, testing for convergence
289  * in the linear or quadratic case.
290  * Initiates one of the variable shift iterations and returns
291  * the number of zeros found.
292  */
293 template<typename Real>
294  int
295  JenkinsTraubSolver<Real>::fxshfr(int l2)
296  {
297  Real ts_old{}, tv_old{};
298  int iflag;
299 
300  int num_zeros = 0;
301 
302  auto betav = Real{0.25};
303  auto betas = Real{0.25};
304  auto ss_old = this->m_sr;
305  auto vv_old = this->m_v;
306  // Evaluate polynomial by synthetic division.
307  this->remquo_quadratic(this->m_order, this->m_u, this->m_v,
308  this->m_P, this->m_P_quot,
309  this->m_a, this->m_b);
310  auto type = this->init_next_h_poly();
311  for (int j = 0; j < l2; ++j)
312  {
313  // Calculate next H polynomial and estimate v.
314  this->next_h_poly(type);
315  type = this->init_next_h_poly();
316  auto [ui, vi] = this->quadratic_coefficients(type);
317  auto vv = vi;
318  // Estimate s.
319  auto ss = Real{0};
320  if (this->m_H[this->m_order - 1] != Real{0})
321  ss = -this->m_P[this->m_order] / this->m_H[this->m_order - 1];
322  auto tv = Real{1};
323  auto ts = Real{1};
324  if (j == 0 || type == near_h_root)
325  {
326  vv_old = vv;
327  ss_old = ss;
328  tv_old = tv;
329  ts_old = ts;
330  continue;
331  }
332 
333  // Compute relative measures of convergence of s and v sequences.
334  if (vv != Real{0})
335  tv = std::abs((vv - vv_old) / vv);
336  if (ss != Real{0})
337  ts = std::abs((ss - ss_old) / ss);
338 
339  // If decreasing, multiply two most recent convergence measures.
340  const auto tvv = tv < tv_old ? tv * tv_old : Real{1};
341  const auto tss = ts < ts_old ? ts * ts_old : Real{1};
342 
343  // Compare with convergence criteria.
344  const auto vpass = tvv < betav;
345  const auto spass = tss < betas;
346  if (!(spass || vpass))
347  {
348  vv_old = vv;
349  ss_old = ss;
350  tv_old = tv;
351  ts_old = ts;
352  continue;
353  }
354 
355  // At least one sequence has passed the convergence test.
356  // Store variables before iterating.
357  const auto u_save = this->m_u;
358  const auto v_save = this->m_v;
359  this->m_H_save = this->m_H;
360  const auto s = ss;
361 
362  // Choose iteration according to the fastest converging sequence.
363  auto vtry = false;
364  auto stry = false;
365  if ((spass && !vpass) || tss < tvv)
366  goto TRY_LINEAR;
367 
368  TRY_QUADRATIC:
369  num_zeros = this->iter_quadratic(ui, vi);
370  if (num_zeros > 0)
371  return num_zeros;
372  // Quadratic iteration has failed. Flag that it has
373  // been tried and decrease the convergence criterion.
374  vtry = true;
375  betav *= Real{0.25};
376  // Try linear iteration if it has not been tried and
377  // the S sequence is converging.
378  if (stry || !spass)
379  goto RESTORE_VARS;
380  this->m_H = this->m_H_save;
381 
382  TRY_LINEAR:
383  num_zeros = this->iter_real(s, iflag);
384  if (num_zeros > 0)
385  return num_zeros;
386  // Linear iteration has failed. Flag that it has been
387  // tried and decrease the convergence criterion.
388  stry = true;
389  betas *= Real{0.25};
390  if (iflag == 0)
391  goto RESTORE_VARS;
392  // If linear iteration signals an almost real
393  // zero attempt quadratic iteration.
394  ui = -Real{2} * s;
395  vi = s * s;
396  goto TRY_QUADRATIC;
397 
398  RESTORE_VARS:
399  // Restore variables.
400  this->m_u = u_save;
401  this->m_v = v_save;
402  this->m_H = this->m_H_save;
403 
404  // Try quadratic iteration if it has not been tried
405  // and the V sequence is converging.
406  if (!vtry && vpass)
407  goto TRY_QUADRATIC;
408 
409  // Recompute polynomial quotient and remainder
410  // to continue the second stage.
411  this->remquo_quadratic(this->m_order, this->m_u, this->m_v,
412  this->m_P, this->m_P_quot,
413  this->m_a, this->m_b);
414  type = this->init_next_h_poly();
415  }
416  return num_zeros;
417  }

420 /**
421  * Variable-shift H-polynomial iteration for a quadratic factor
422  * converges only if the zeros are equimodular or nearly so.
423  * @param uu The linear coefficient of the starting quadratic equation.
424  * @param vv The constant coefficient of the starting quadratic equation.
425  * @return The number of zeros found.
426  */
427 template<typename Real>
428  int
429  JenkinsTraubSolver<Real>::iter_quadratic(Real uu, Real vv)
430  {
431  Real mp, mp_old{}, ee, relstp = std::sqrt(s_eps), t, zm;
432  NormalizationType type;
433 
434  int num_zeros = 0;
435  bool tried = false;
436  this->m_u = uu;
437  this->m_v = vv;
438  int j = 0;
439 
440  while (true)
441  {
442  ++this->m_num_iters;
443  this->quadratic(Real{1}, this->m_u, this->m_v,
444  this->m_z_small, this->m_z_large);
445  // Return if roots of the quadratic are real and not
446  // close to multiple or nearly equal and of opposite sign.
447  if (std::abs(std::abs(real(this->m_z_small))
448  - std::abs(real(this->m_z_large)))
449  > Real{0.01L} * std::abs(real(this->m_z_large)))
450  return num_zeros;
451  // Evaluate polynomial by quadratic synthetic division.
452  this->remquo_quadratic(this->m_order, this->m_u, this->m_v,
453  this->m_P, this->m_P_quot, this->m_a, this->m_b);
454  mp = std::abs(this->m_a - real(this->m_z_small) * this->m_b)
455  + std::abs(imag(this->m_z_small) * this->m_b);
456  // Compute a rigorous bound on the rounding error in evaluating _P.
457  zm = std::sqrt(std::abs(this->m_v));
458  ee = Real{2} * std::abs(this->m_P_quot[0]);
459  t = -real(this->m_z_small) * this->m_b;
460  for (int i = 1; i < this->m_order; ++i)
461  ee = ee * zm + std::abs(this->m_P_quot[i]);
462  ee = ee * zm + std::abs(this->m_a + t);
463  ee *= (Real{5} * this->m_mre + Real{4} * this->m_are);
464  ee -= (Real{5} * this->m_mre + Real{2} * this->m_are)
465  * (std::abs(this->m_a + t) + std::abs(this->m_b) * zm);
466  ee += Real{2} * this->m_are * std::abs(t);
467  // Iteration has converged sufficiently if the
468  // polynomial value is less than 20 times this bound.
469  if (mp <= Real{20} * ee)
470  {
471  num_zeros = 2;
472  return num_zeros;
473  }
474  ++j;
475  // Stop iteration after 20 steps.
476  if (j > this->m_max_iter_quadratic)
477  return num_zeros;
478  if (j < 2 || relstp > Real{0.01L} || mp < mp_old || tried)
479  {
480  mp_old = mp;
481  // Calculate next H polynomial and new u and v.
482  type = this->init_next_h_poly();
483  this->next_h_poly(type);
484  type = this->init_next_h_poly();
485  const auto [ui, vi] = this->quadratic_coefficients(type);
486  // If vi is zero the iteration is not converging.
487  if (vi == Real{0})
488  return num_zeros;
489  relstp = std::abs((vi - this->m_v) / vi);
490  this->m_u = ui;
491  this->m_v = vi;
492  continue;
493  }
494  // A cluster appears to be stalling the convergence.
495  // Five fixed shift steps are taken with a u, v close to the cluster.
496  if (relstp < s_eps)
497  relstp = s_eps;
498  relstp = std::sqrt(relstp);
499  this->m_u -= this->m_u * relstp;
500  this->m_v += this->m_v * relstp;
501  this->remquo_quadratic(this->m_order, this->m_u, this->m_v,
502  this->m_P, this->m_P_quot,
503  this->m_a, this->m_b);
504  for (int i = 0; i < 5; ++i)
505  {
506  type = this->init_next_h_poly();
507  this->next_h_poly(type);
508  }
509  tried = true;
510  j = 0;
511  }
512  }

515 /**
516  * Variable-shift H polynomial iteration for a real zero.
517  * @param sss Starting iterate.
518  * @param iflag Flag to indicate a pair of zeros near real axis.
519  * @return The number of zeros found.
520  */
521 template<typename Real>
522  int
523  JenkinsTraubSolver<Real>::iter_real(Real sss, int& iflag)
524  {
525  auto t = Real{0};
526  decltype(std::abs(this->m_P[0])) mp_old{};
527 
528  int num_zeros = 0;
529  auto s = sss;
530  iflag = 0;
531  int i_real = 0;
532 
533  while (true)
534  {
535  ++this->m_num_iters;
536  auto pval = this->m_P[0];
537  // Evaluate P at s.
538  this->m_P_quot[0] = pval;
539  for (int i = 1; i <= this->m_order; ++i)
540  {
541  pval = pval * s + this->m_P[i];
542  this->m_P_quot[i] = pval;
543  }
544  auto mp = std::abs(pval);
545  // Compute a rigorous bound on the error in evaluating P.
546  const auto ms = std::abs(s);
547  auto ee = (this->m_mre / (this->m_are + this->m_mre))
548  * std::abs(this->m_P_quot[0]);
549  for (int i = 1; i <= this->m_order; ++i)
550  ee = ee * ms + std::abs(this->m_P_quot[i]);
551  // Iteration has converged sufficiently if the polynomial
552  // value is less than 20 times this bound.
553  if (mp <= Real{20}
554  * ((this->m_are + this->m_mre) * ee - this->m_mre * mp))
555  {
556  num_zeros = 1;
557  this->m_z_small = s;
558  return num_zeros;
559  }
560  ++i_real;
561  // Stop iteration after max_iter_real steps.
562  if (i_real > this->m_max_iter_real)
563  return num_zeros;
564  else if (i_real < 2
565  || std::abs(t) > Real{0.001L} * std::abs(s - t)
566  || mp < mp_old)
567  {
568  // Return if the polynomial value has increased significantly.
569  mp_old = mp;
570 
571  // Compute t, the next polynomial, and the new iterate.
572  auto hval = this->m_H[0];
573  this->m_H_quot[0] = hval;
574  for (int i = 1; i < this->m_order; ++i)
575  {
576  hval = hval * s + this->m_H[i];
577  this->m_H_quot[i] = hval;
578  }
579 
580  if (std::abs(hval)
581  <= std::abs(this->m_H[this->m_order - 1]) * Real{10} * s_eps)
582  {
583  // Use unscaled form.
584  this->m_H[0] = Real{0};
585  for (int i = 1; i < this->m_order; ++i)
586  this->m_H[i] = this->m_H_quot[i-1];
587  }
588  else
589  {
590  // Use the scaled form of the recurrence if the value
591  // of H at s is nonzero.
592  t = -pval / hval;
593  this->m_H[0] = this->m_P_quot[0];
594  for (int i = 1; i < this->m_order; ++i)
595  this->m_H[i] = t * this->m_H_quot[i - 1]
596  + this->m_P_quot[i];
597  }
598 
599  hval = this->m_H[0];
600  for (int i = 1; i < this->m_order; ++i)
601  hval = hval * s + this->m_H[i];
602  auto t = Real{0};
603  if (std::abs(hval)
604  > std::abs(this->m_H[this->m_order - 1] * Real{10} * s_eps))
605  t = -pval / hval;
606  s += t;
607  }
608  else
609  {
610  // A cluster of zeros near the real axis has been encountered.
611  // Return with iflag set to initiate a quadratic iteration.
612  iflag = 1;
613  sss = s;
614  return num_zeros;
615  }
616  }

619 /**
620  * This routine calculates scalar quantities used to compute
621  * the next H-polynomial and new estimates of the quadratic coefficients.
622  *
623  * @return Flag indicating how the calculations are normalized
624  * to avoid overflow.
625  */
626 template<typename Real>
627  typename JenkinsTraubSolver<Real>::NormalizationType
628  JenkinsTraubSolver<Real>::init_next_h_poly()
629  {
630  const auto eps = Real{100} * s_eps;
631  // Synthetic division of H by the quadratic 1, u, v
632  NormalizationType type = none;
633  this->remquo_quadratic(this->m_order - 1, this->m_u, this->m_v,
634  this->m_H, this->m_H_quot, this->m_c, this->m_d);
635  if (std::abs(this->m_c) > eps * std::abs(this->m_H[this->m_order - 1])
636  || std::abs(this->m_d) > eps * std::abs(this->m_H[this->m_order - 2]))
637  {
638  if (std::abs(this->m_d) < std::abs(this->m_c))
639  {
640  type = divide_by_c;
641  this->m_e = this->m_a / this->m_c;
642  this->m_f = this->m_d / this->m_c;
643  this->m_g = this->m_u * this->m_e;
644  this->m_h = this->m_v * this->m_b;
645  this->m_a3 = this->m_a * this->m_e
646  + (this->m_h / this->m_c + this->m_g) * this->m_b;
647  this->m_a1 = this->m_b - this->m_a * (this->m_d / this->m_c);
648  this->m_a7 = this->m_a
649  + this->m_g * this->m_d
650  + this->m_h * this->m_f;
651  return type;
652  }
653  else
654  {
655  type = divide_by_d;
656  this->m_e = this->m_a / this->m_d;
657  this->m_f = this->m_c / this->m_d;
658  this->m_g = this->m_u * this->m_b;
659  this->m_h = this->m_v * this->m_b;
660  this->m_a3 = (this->m_a + this->m_g) * this->m_e
661  + this->m_h * (this->m_b / this->m_d);
662  this->m_a1 = this->m_b * this->m_f - this->m_a;
663  this->m_a7 = (this->m_f + this->m_u) * this->m_a + this->m_h;
664  return type;
665  }
666  }
667  else
668  {
669  type = near_h_root;
670  return type;
671  }
672  }

675 /**
676  * Computes the next H polynomials using scalars computed in init_next_h_poly.
677  */
678 template<typename Real>
679  void
680  JenkinsTraubSolver<Real>::next_h_poly(NormalizationType type)
681  {
682  if (type == near_h_root)
683  {
684  // Use unscaled form of the recurrence if type is 3.
685  this->m_H[0] = Real{0};
686  this->m_H[1] = Real{0};
687  for (int i = 2; i < this->m_order; ++i)
688  this->m_H[i] = this->m_H_quot[i-2];
689  return;
690  }
691  auto ab_temp = this->m_a;
692  if (type == divide_by_c)
693  ab_temp = this->m_b;
694  if (std::abs(this->m_a1) <= std::abs(ab_temp) * s_eps * Real{10})
695  {
696  // If a1 is nearly zero then use a special form of the recurrence.
697  this->m_H[0] = Real{0};
698  this->m_H[1] = -this->m_a7 * this->m_P_quot[0];
699  for(int i = 2; i < this->m_order; ++i)
700  this->m_H[i] = this->m_a3 * this->m_H_quot[i-2]
701  - this->m_a7 * this->m_P_quot[i-1];
702  }
703  else
704  {
705  // Use scaled form of the recurrence.
706  this->m_a7 /= this->m_a1;
707  this->m_a3 /= this->m_a1;
708  this->m_H[0] = this->m_P_quot[0];
709  this->m_H[1] = this->m_P_quot[1] - this->m_a7 * this->m_P_quot[0];
710  for (int i = 2; i < this->m_order; ++i)
711  this->m_H[i] = this->m_a3 * this->m_H_quot[i-2]
712  - this->m_a7 * this->m_P_quot[i-1]
713  + this->m_P_quot[i];
714  }
715  }

717 /**
718  * Compute new estimates of the quadratic coefficients
719  * using the scalars computed in init_next_h_poly.
720  */
721 template<typename Real>
722  std::pair<Real, Real>
723  JenkinsTraubSolver<Real>::quadratic_coefficients(NormalizationType type)
724  {
725  if (type == near_h_root)
726  return std::make_pair(Real{0}, Real{0});
727 
728  Real a4, a5;
729  if (type == divide_by_d)
730  {
731  a4 = (this->m_a + this->m_g) * this->m_f + this->m_h;
732  a5 = (this->m_f + this->m_u) * this->m_c + this->m_v * this->m_d;
733  }
734  else
735  {
736  a4 = this->m_a + this->m_u * this->m_b + this->m_h * this->m_f;
737  a5 = this->m_c + (this->m_u + this->m_v * this->m_f) * this->m_d;
738  }
739 
740  // Evaluate new quadratic coefficients.
741  const auto n = this->m_order;
742  const auto b1 = -this->m_H[n - 1] / this->m_P[n];
743  const auto b2 = -(this->m_H[n - 2] + b1 * this->m_P[n - 1]) / this->m_P[n];
744  const auto c1 = this->m_v * b2 * this->m_a1;
745  const auto c2 = b1 * this->m_a7;
746  const auto c3 = b1 * b1 * this->m_a3;
747  const auto c4 = c1 - c2 - c3;
748  if (auto temp = a5 + b1 * a4 - c4; temp == Real{0})
749  return std::make_pair(Real{0}, Real{0});
750  else
751  {
752  auto uu = this->m_u - (this->m_u * (c3 + c2)
753  + this->m_v * (b1 * this->m_a1 + b2 * this->m_a7)) / temp;
754  auto vv = this->m_v * (Real{1} + c4 / temp);
755  return std::make_pair(uu, vv);
756  }
757  }

760  * Divides the polynomial P by the quadratic 1x^2 + ux + v
761  * placing the quotient in q and the remainder in a, b.
762  */
763 template<typename Real>
764  void
765  JenkinsTraubSolver<Real>::remquo_quadratic(int nn, Real u, Real v,
766  std::vector<Real>& poly,
767  std::vector<Real>& quot,
768  Real& a, Real& b)
769  {
770  b = poly[0];
771  quot[0] = b;
772  a = poly[1] - b * u;
773  quot[1] = a;
774  for (int i = 2; i <= nn; ++i)
775  {
776  auto c = poly[i] - a * u - b * v;
777  quot[i] = c;
778  b = a;
779  a = c;
780  }
781  }

784 /**
785  * Calculate the zeros of the quadratic az^2 + bz + c.
786  * The quadratic formula, modified to avoid overflow, is used
787  * to find the larger zero if the zeros are real and both
788  * are complex. The smaller real zero is found directly from
789  * the product of the zeros c/a.
790  */
791 template<typename Real>
792  void
793  JenkinsTraubSolver<Real>::quadratic(Real a, Real b, Real c,
794  Solution<Real>& z_small,
795  Solution<Real>& z_large)
796  {
797  z_small = {};
798  z_large = {};
799  if (a == Real{0})
800  { // Less than two roots.
801  if (b != Real{0})
802  z_small = -c / b;
803  return;
804  }
805 
806  if (c == Real{0})
807  { // one real root, one zero root.
808  z_small = Real{0};
809  z_large = -b / a;
810  return;
811  }
812 
813  // Compute discriminant avoiding overflow.
814  auto b2 = b / Real{2};
815 
816  Real d, e;
817  if (std::abs(b2) < std::abs(c))
818  {
819  e = std::copysign(a, c);
820  e = b2 * (b2 / std::abs(c)) - e;
821  d = std::sqrt(std::abs(e)) * std::sqrt(std::abs(c));
822  }
823  else
824  {
825  e = Real{1} - (a / b2) * (c / b2);
826  d = std::sqrt(std::abs(e)) * std::abs(b2);
827  }
828 
829  if (e < Real{0})
830  { // complex conjugate zeros.
831  z_small = std::complex<Real>(-b2 / a, +std::abs(d / a));
832  z_large = std::complex<Real>(-b2 / a, -std::abs(d / a));
833  }
834  else
835  {
836  if (b2 >= Real{0})
837  d = -d; // Real zeros.
838  z_large = (-b2 + d) / a;
839  z_small = Real{0};
840  if (z_large != Real{0})
841  z_small = (c / z_large) / a;
842  }
843  }
