76  /**
77  *
78  template<typename Tp>
79  //std::enable_if_t<std::is_floating_point_v<Tp>>
80  void
81  Polynomial<Tp>::m_set_scale()
82  {
83  constexpr real_type s_eps = std::numeric_limits<real_type>::epsilon();
84  constexpr real_type
85  s_base = real_type{std::numeric_limits<real_type>::radix};
86  constexpr real_type s_tiny = s_eps * s_eps * s_eps; // ~ 1.0e-50;
87  constexpr real_type s_huge = std::numeric_limits<real_type>::max();
88  constexpr real_type s_low = s_tiny / s_eps;
89 
90  // Find largest and smallest moduli of coefficients.
91  auto a_max = real_type{0};
92  auto a_min = s_huge;
93  for (int i = 0; i <= this->degree(); ++i)
94  {
95  const auto x = get_scale(this->m_coeff[i]);
96  if (x > a_max)
97  a_max = x;
98  if (x != real_type{0} && x < a_min)
99  a_min = x;
100  }
101  // Scale if there are large or very tiny coefficients.
102  // Computes a scale factor to multiply the coefficients
103  // of the polynomial. The scaling is done to avoid overflow
104  // and to avoid undetected underflow interfering
105  // with the convergence criterion.
106  // The factor is a power of the base.
107  auto scale = s_low / a_min;
108  bool rescale = true;
109  if (scale > real_type{1} && s_huge / scale < a_max)
110  rescale = false;
111  if (scale <= real_type{1} && a_max < real_type{10})
112  rescale = false;
113 
114  this->m_scale = real_type{1};
115  if (rescale)
116  {
117  // Scale polynomial.
118  if (scale == real_type{0})
119  scale = s_tiny;
120  const auto lg = std::ilogb(scale);
121  this->m_scale = std::pow(s_base, lg);
122  //if (this->m_scale != real_type{1})

140  /**
141  * Evaluate the polynomial using a modification of Horner's rule which
142  * exploits the fact that the polynomial coefficients are all real.
143  *
144  * The algorithm is discussed in detail in:
145  * Knuth, D. E., The Art of Computer Programming: Seminumerical
146  * Algorithms (Vol. 2) Third Ed., Addison-Wesley, pp 486-488, 1998.
147  *
148  * If n is the degree of the polynomial,
149  * n - 3 multiplies and 4 * n - 6 additions are saved.
150  */
151  template<typename Tp>
152  template<typename Up>
153  auto
154  Polynomial<Tp>::operator()(const std::complex<Up>& z) const
155  -> std::enable_if_t<!has_imag_v<Tp>,
156  std::complex<std::decay_t<
157  decltype(typename Polynomial<Tp>::value_type{} * Up{})>>>
158  {
159  const auto r = Tp{2} * std::real(z);
160  const auto s = std::norm(z);
161  size_type n = this->degree();
162  auto aa = this->coefficient(n);
163  if (n > 0)
164  {
165  auto bb = this->coefficient(n - 1);
166  for (size_type j = 2; j <= n; ++j)
167  bb = this->coefficient(n - j)
168  - s * std::exchange(aa, bb + r * aa);
169  return aa * z + bb;
170  }
171  else
172  return aa;
173  };

175  // Could/should this be done by output iterator range?
176  template<typename Tp>
177  template<typename Polynomial<Tp>::size_type N>
178  void
179  Polynomial<Tp>::eval(typename Polynomial<Tp>::value_type x,
180  std::array<Polynomial<Tp>::value_type, N>& arr)
181  {
182  if (arr.size() > 0)
183  {
184  arr.fill(value_type{});
185  const size_type sz = m_coeff.size();
186  arr[0] = this->coefficient(sz - 1);
187  for (int i = sz - 2; i >= 0; --i)
188  {
189  int nn = std::min(arr.size() - 1, sz - 1 - i);
190  for (int j = nn; j >= 1; --j)
191  arr[j] = std::fma(arr[j], x, arr[j - 1]);
192  arr[0] = std::fma(arr[0], x, this->coefficient(i));
193  }
194  // Now put in the factorials.
195  value_type fact = value_type(1);
196  for (size_type n = arr.size(), i = 2; i < n; ++i)
197  {
198  fact *= value_type(i);
199  arr[i] *= fact;
200  }
201  }
202  }

204  /**
205  * Evaluate the polynomial and its derivatives at the point x.
206  * The values are placed in the output range starting with the
207  * polynomial value and continuing through higher derivatives.
208  */
209  template<typename Tp>
210  template<typename OutIter>
211  void
212  Polynomial<Tp>::eval(typename Polynomial<Tp>::value_type x,
213  OutIter b, OutIter e)
214  {
215  if(b != e)
216  {
217  std::fill(b, e, value_type{});
218  const size_type sz = m_coeff.size();
219  *b = m_coeff[sz - 1];
220  for (int i = sz - 2; i >= 0; --i)
221  {
222  for (auto it = std::reverse_iterator<OutIter>(e);
223  it != std::reverse_iterator<OutIter>(b) - 1; ++it)
224  *it = std::fma(*it, x, *(it + 1));
225  *b = std::fma(*b, x, m_coeff[i]);
226  }
227  // Now put in the factorials.
228  int i = 0;
229  value_type fact = value_type(++i);
230  for (auto it = b + 1; it != e; ++it)
231  {
232  fact *= value_type(i);
233  *it *= fact;
234  ++i;
235  }
236  }
237  }

239  /**
240  * Evaluate the even part of the polynomial at the input point.
241  */
242  template<typename Tp>
243  typename Polynomial<Tp>::value_type
244  Polynomial<Tp>::eval_even(typename Polynomial<Tp>::value_type x) const
245  {
246  if (this->degree() > 0)
247  {
248  const auto odd = this->degree() % 2;
249  const auto xx = x * x;
250  auto poly(this->coefficient(this->degree() - odd));
251  for (int i = this->degree() - odd - 2; i >= 0; i -= 2)
252  poly = std::fma(xx, poly, this->coefficient(i));
253  return poly;
254  }
255  else
256  return value_type{};
257  }

279  /**
280  * Evaluate the even part of the polynomial using a modification
281  * of Horner's rule which exploits the fact that the polynomial
282  * coefficients are all real.
283  *
284  * The algorithm is discussed in detail in:
285  * Knuth, D. E., The Art of Computer Programming: Seminumerical
286  * Algorithms (Vol. 2) Third Ed., Addison-Wesley, pp 486-488, 1998.
287  *
288  * If n is the degree of the polynomial,
289  * n - 3 multiplies and 4 * n - 6 additions are saved.
290  */
291  template<typename Tp>
292  template<typename Up>
293  auto
294  Polynomial<Tp>::eval_even(const std::complex<Up>& z) const
295  -> std::enable_if_t<!has_imag_v<Tp>,
296  std::complex<std::decay_t<
297  decltype(typename Polynomial<Tp>::value_type{} * Up{})>>>
298  {
299  using real_t = std::decay_t<decltype(value_type{} * Up{})>;
300  using cmplx_t = std::complex<real_t>;
301  if (this->degree() > 0)
302  {
303  const auto zz = z * z;
304  const auto r = Tp{2} * std::real(zz);
305  const auto s = std::norm(zz);
306  auto odd = this->degree() % 2;
307  size_type n = this->degree() - odd;
308  auto aa = this->coefficient(n);
309  auto bb = this->coefficient(n - 2);
310  for (size_type j = 4; j <= n; j += 2)
311  bb = std::fma(-s, std::exchange(aa, bb + r * aa),
312  this->coefficient(n - j));
313  return emsr::fma(cmplx_t(aa), cmplx_t(zz), cmplx_t(bb));
314  }
315  else
316  return cmplx_t{};
317  };

361  * Multiply the polynomial by another polynomial.

363  template<typename Tp>
364  template<typename Up>
365  Polynomial<Tp>&
366  Polynomial<Tp>::operator*=(const Polynomial<Up>& poly)
367  {
368  // Test for zero size polys and do special processing?
369  const size_type m = this->degree();
370  const size_type n = poly.degree();
371  std::vector<value_type> new_coeff(m + n + 1);
372  for (size_type i = 0; i <= m; ++i)
373  for (size_type j = 0; j <= n; ++j)

376  this->m_coeff = new_coeff;

380  /**
381  * Divide two polynomials returning the quotient and remainder.
382  */
383  template<typename Tp>
384  void
385  divmod(const Polynomial<Tp>& num, const Polynomial<Tp>& den,
386  Polynomial<Tp>& quo, Polynomial<Tp>& rem)
387  {
388  rem = num;
389  quo = Polynomial<Tp>(Tp{}, num.degree());
390  const std::size_t d_num = num.degree();
391  const std::size_t d_den = den.degree();
392  if (d_den <= d_num)
393  {
394  for (int k = d_num - d_den; k >= 0; --k)
395  {
396  quo.coefficient(k, rem.coefficient(d_den + k)
397  / den.coefficient(d_den));
398  for (int j = d_den + k - 1; j >= k; --j)
399  rem.coefficient(j, rem.coefficient(j)
400  - quo.coefficient(k)
401  * den.coefficient(j - k));
402  }
403  quo.degree(d_num - d_den);
404  rem.degree(d_den > 0 ? d_den - 1 : 0);
405  }
406  else
407  quo.degree(0);
408  }

410  /**
411  * Write a polynomial to a stream.
412  * The format is a parenthesized comma-delimited list of coefficients.
413  */
414  template<typename CharT, typename Traits, typename Tp>
415  std::basic_ostream<CharT, Traits>&
416  operator<<(std::basic_ostream<CharT, Traits>& os,
417  const Polynomial<Tp>& poly)
418  {
419  int old_prec = os.precision(std::numeric_limits<Tp>::max_digits10);
420  os << "(";
421  for (size_t i = 0; i < poly.degree(); ++i)
422  os << poly.coefficient(i) << ",";
423  os << poly.coefficient(poly.degree());
424  os << ")";
425  os.precision(old_prec);
426  return os;
427  }

429  /**
430  * Read a polynomial from a stream.
431  * The input format can be a plain scalar (zero degree polynomial)
432  * or a parenthesized comma-delimited list of coefficients.
433  */
434  template<typename CharT, typename Traits, typename Tp>
435  std::basic_istream<CharT, Traits>&
436  operator>>(std::basic_istream<CharT, Traits>& is,
437  Polynomial<Tp>& poly)
438  {
439  Tp x;
440  CharT ch;
441  is >> ch;
442  if (ch == '(')
443  {
444  do
445  {
446  is >> x >> ch;
447  poly.m_coeff.push_back(x);
448  }
449  while (ch == ',');
450  if (ch != ')')
451  is.setstate(std::ios_base::failbit);
452  }
453  else
454  {
455  is.putback(ch);
456  is >> x;
457  poly = x;
458  }
459  // No null polynomial.
460  if (poly.size() == 0)
461  poly.m_coeff.resize(1, Tp{});
462 
463  return is;
464  }
