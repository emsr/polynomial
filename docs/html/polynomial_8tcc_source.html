86  template<typename _Tp>
87  //std::enable_if_t<std::is_floating_point_v<_Tp>>
88  void
89  _Polynomial<_Tp>::_M_set_scale()
90  {
91  constexpr real_type _S_eps = std::numeric_limits<real_type>::epsilon();
92  constexpr real_type
93  _S_base = real_type{std::numeric_limits<real_type>::radix};
94  constexpr real_type _S_tiny = _S_eps * _S_eps * _S_eps; // ~ 1.0e-50;
95  constexpr real_type _S_huge = std::numeric_limits<real_type>::max();
96  constexpr real_type _S_low = _S_tiny / _S_eps;
97 
98  // Find largest and smallest moduli of coefficients.
99  auto __a_max = real_type{0};
100  auto __a_min = _S_huge;
101  for (int __i = 0; __i <= this->degree(); ++__i)
102  {
103  const auto __x = get_scale(this->_M_coeff[__i]);
104  if (__x > __a_max)
105  __a_max = __x;
106  if (__x != real_type{0} && __x < __a_min)
107  __a_min = __x;
108  }
109  // Scale if there are large or very tiny coefficients.
110  // Computes a scale factor to multiply the coefficients
111  // of the polynomial. The scaling is done to avoid overflow
112  // and to avoid undetected underflow interfering
113  // with the convergence criterion.
114  // The factor is a power of the base.
115  auto __scale = _S_low / __a_min;
116  bool __rescale = true;
117  if (__scale > real_type{1} && _S_huge / __scale < __a_max)
118  __rescale = false;
119  if (__scale <= real_type{1} && __a_max < real_type{10})
120  __rescale = false;
121 
122  this->_M_scale = real_type{1};
123  if (__rescale)
124  {
125  // Scale polynomial.
126  if (__scale == real_type{0})
127  __scale = _S_tiny;
128  const auto __lg = std::ilogb(__scale);
129  this->_M_scale = std::pow(_S_base, __lg);

148  /**
149  * Evaluate the polynomial using a modification of Horner's rule which
150  * exploits the fact that the polynomial coefficients are all real.
151  *
152  * The algorithm is discussed in detail in:
153  * Knuth, D. E., The Art of Computer Programming: Seminumerical
154  * Algorithms (Vol. 2) Third Ed., Addison-Wesley, pp 486-488, 1998.
155  *
156  * If n is the degree of the polynomial,
157  * n - 3 multiplies and 4 * n - 6 additions are saved.
158  */
159  template<typename _Tp>
160  template<typename _Up>
161  auto
162  _Polynomial<_Tp>::operator()(const std::complex<_Up>& __z) const
163  -> std::enable_if_t<!__has_imag_v<_Tp>,
164  std::complex<std::decay_t<
165  decltype(typename _Polynomial<_Tp>::value_type{} * _Up{})>>>
166  {
167  const auto __r = _Tp{2} * std::real(__z);
168  const auto __s = std::norm(__z);
169  size_type __n = this->degree();
170  auto __aa = this->coefficient(__n);
171  if (__n > 0)
172  {
173  auto __bb = this->coefficient(__n - 1);
174  for (size_type __j = 2; __j <= __n; ++__j)
175  __bb = this->coefficient(__n - __j)
176  - __s * std::exchange(__aa, __bb + __r * __aa);
177  return __aa * __z + __bb;
178  }
179  else
180  return __aa;
181  };

186  void
187  _Polynomial<_Tp>::eval(typename _Polynomial<_Tp>::value_type __x,
188  std::array<_Polynomial<_Tp>::value_type, N>& __arr)
189  {
190  if (__arr.size() > 0)
191  {
192  __arr.fill(value_type{});
193  const size_type __sz = _M_coeff.size();
194  __arr[0] = this->coefficient(__sz - 1);
195  for (int __i = __sz - 2; __i >= 0; --__i)
196  {
197  int __nn = std::min(__arr.size() - 1, __sz - 1 - __i);
198  for (int __j = __nn; __j >= 1; --__j)
199  __arr[__j] = std::fma(__arr[__j], __x, __arr[__j - 1]);
200  __arr[0] = std::fma(__arr[0], __x, this->coefficient(__i));
201  }
202  // Now put in the factorials.
203  value_type __fact = value_type(1);
204  for (size_type __n = __arr.size(), __i = 2; __i < __n; ++__i)
205  {
206  __fact *= value_type(__i);
207  __arr[__i] *= __fact;
208  }
209  }

213  * Evaluate the polynomial and its derivatives at the point x.
214  * The values are placed in the output range starting with the
215  * polynomial value and continuing through higher derivatives.
216  */

218  template<typename OutIter>
219  void
220  _Polynomial<_Tp>::eval(typename _Polynomial<_Tp>::value_type __x,
221  OutIter __b, OutIter __e)
222  {
223  if(__b != __e)
224  {
225  std::fill(__b, __e, value_type{});
226  const size_type __sz = _M_coeff.size();
227  *__b = _M_coeff[__sz - 1];
228  for (int __i = __sz - 2; __i >= 0; --__i)
229  {
230  for (auto __it = std::reverse_iterator<OutIter>(__e);
231  __it != std::reverse_iterator<OutIter>(__b) - 1; ++__it)
232  *__it = std::fma(*__it, __x, *(__it + 1));
233  *__b = std::fma(*__b, __x, _M_coeff[__i]);

235  // Now put in the factorials.

237  value_type __fact = value_type(++__i);

239  {

243  }

299  template<typename _Tp>
300  template<typename _Up>
301  auto
302  _Polynomial<_Tp>::eval_even(const std::complex<_Up>& __z) const
303  -> std::enable_if_t<!__has_imag_v<_Tp>,
304  std::complex<std::decay_t<
305  decltype(typename _Polynomial<_Tp>::value_type{} * _Up{})>>>
306  {
307  using __real_t = std::decay_t<decltype(value_type{} * _Up{})>;
308  using __cmplx_t = std::complex<__real_t>;
309  if (this->degree() > 0)
310  {
311  const auto __zz = __z * __z;
312  const auto __r = _Tp{2} * std::real(__zz);
313  const auto __s = std::norm(__zz);
314  auto __odd = this->degree() % 2;
315  size_type __n = this->degree() - __odd;

317  auto __bb = this->coefficient(__n - 2);
318  for (size_type __j = 4; __j <= __n; __j += 2)

388  /**
389  * Divide two polynomials returning the quotient and remainder.
390  */
391  template<typename _Tp>
392  void
393  divmod(const _Polynomial<_Tp>& __num, const _Polynomial<_Tp>& __den,
394  _Polynomial<_Tp>& __quo, _Polynomial<_Tp>& __rem)
395  {
396  __rem = __num;
397  __quo = _Polynomial<_Tp>(_Tp{}, __num.degree());
398  const std::size_t __d_num = __num.degree();
399  const std::size_t __d_den = __den.degree();
400  if (__d_den <= __d_num)
401  {
402  for (int __k = __d_num - __d_den; __k >= 0; --__k)
403  {
404  __quo.coefficient(__k, __rem.coefficient(__d_den + __k)
405  / __den.coefficient(__d_den));
406  for (int __j = __d_den + __k - 1; __j >= __k; --__j)
407  __rem.coefficient(__j, __rem.coefficient(__j)
408  - __quo.coefficient(__k)
409  * __den.coefficient(__j - __k));
410  }
411  __quo.degree(__d_num - __d_den);
412  __rem.degree(__d_den > 0 ? __d_den - 1 : 0);
413  }
414  else
415  __quo.degree(0);
416  }

438  * Read a polynomial from a stream.
439  * The input format can be a plain scalar (zero degree polynomial)
440  * or a parenthesized comma-delimited list of coefficients.
441  */
442  template<typename CharT, typename Traits, typename _Tp>
443  std::basic_istream<CharT, Traits>&
444  operator>>(std::basic_istream<CharT, Traits>& __is,
445  _Polynomial<_Tp>& __poly)
446  {
447  _Tp __x;
448  CharT __ch;
449  __is >> __ch;
450  if (__ch == '(')
451  {
452  do
453  {
454  __is >> __x >> __ch;
455  __poly._M_coeff.push_back(__x);
456  }
457  while (__ch == ',');
458  if (__ch != ')')
459  __is.setstate(std::ios_base::failbit);
460  }
461  else
462  {
463  __is.putback(__ch);
464  __is >> __x;

466  }

469  __poly._M_coeff.resize(1, _Tp{});
