108  /**
109  * Evaluate the polynomial using a modification of Horner's rule which
110  * exploits the fact that the polynomial coefficients are all real.
111  *
112  * The algorithm is discussed in detail in:
113  * Knuth, D. E., The Art of Computer Programming: Seminumerical
114  * Algorithms (Vol. 2) Third Ed., Addison-Wesley, pp 486-488, 1998.
115  *
116  * If n is the degree of the polynomial,
117  * n - 3 multiplies and 4 * n - 6 additions are saved.
118  */
119  template<typename _Tp>
120  template<typename _Up>
121  auto
122  _Polynomial<_Tp>::operator()(const std::complex<_Up>& __z) const
123  -> decltype(_Polynomial<_Tp>::value_type{} * std::complex<_Up>{})
124  {
125  const auto __r = _Tp{2} * std::real(__z);
126  const auto __s = std::norm(__z);
127  size_type __n = this->degree();
128  auto __aa = this->coefficient(__n);
129  if (__n > 0)
130  {
131  auto __bb = this->coefficient(__n - 1);
132  for (size_type __j = 2; __j <= __n; ++__j)
133  __bb = this->coefficient(__n - __j)
134  - __s * std::exchange(__aa, __bb + __r * __aa);
135  return __aa * __z + __bb;
136  }
137  else
138  return __aa;
139  };

349  template<typename _Tp>

351  divmod(const _Polynomial<_Tp>& __pa, const _Polynomial<_Tp>& __pb,
352  _Polynomial<_Tp>& __quo, _Polynomial<_Tp>& __rem)
353  {
354  __rem = __pa;
355  __quo = _Polynomial<_Tp>(_Tp(), __pa.degree());
356  const std::size_t __na = __pa.degree();
357  const std::size_t __nb = __pb.degree();
358  if (__nb <= __na)
359  {
360  for (int __k = __na - __nb; __k >= 0; --__k)
361  {
362  __quo.coefficient(__k, __rem.coefficient(__nb + __k)
363  / __pb.coefficient(__nb));
364  for (int __j = __nb + __k - 1; __j >= __k; --__j)
365  __rem.coefficient(__j, __rem.coefficient(__j)
366  - __quo.coefficient(__k)
367  * __pb.coefficient(__j - __k));
368  }
369  for (int __j = __nb; __j <= __na; ++__j)
370  __rem.coefficient(__j, _Tp());

393  /**
394  * Read a polynomial from a stream.
395  * The input format can be a plain scalar (zero degree polynomial)
396  * or a parenthesized comma-delimited list of coefficients.
397  */
398  template<typename CharT, typename Traits, typename _Tp>
399  std::basic_istream<CharT, Traits>&
400  operator>>(std::basic_istream<CharT, Traits>& __is,
401  _Polynomial<_Tp>& __poly)
402  {
403  _Tp __x;
404  CharT __ch;
405  __is >> __ch;
406  if (__ch == '(')
407  {
408  do
409  {
410  __is >> __x >> __ch;

416  }
417  else
418  {
419  __is.putback(__ch);
420  __is >> __x;
421  __poly = __x;
422  }
423  return __is;
