194  /**
195  * The following polynomial evaluations are done using
196  * a modified of Horner's rule which exploits the fact that
197  * the polynomial coefficients are all real.
198  * The algorithm is discussed in detail in:
199  * Knuth, D. E., The Art of Computer Programming: Seminumerical
200  * Algorithms (Vol. 2) Third Ed., Addison-Wesley, pp 486-488, 1998.
201  *
202  * If n is the degree of the polynomial, n - 3 multiplies are
203  * saved and 4 * n - 6 additions are saved.
204  */
205  template<typename _Tp2>
206  constexpr auto
207  operator()(std::complex<_Tp2> __z) const
208  -> decltype(value_type{} * std::complex<_Tp2>{})
209  {
210  const auto __r = _Tp{2} * std::real(__z);
211  const auto __s = std::norm(__z);
212  auto __aa = this->coefficient(this->degree());
213  auto __bb = this->coefficient(this->degree() - 1);
214  for (int __j = 1; __j <= this->degree(); ++__j)
215  {
216  auto __cc = __s * __aa;
217  __aa = __bb + __r * __aa;
218  __bb = this->coefficient(this->degree() - __j) - __cc;
219  }
220  return __aa * __z + __bb;
221  };

223  /**
224  * Evaluate the polynomial at a range of input points.
225  * The output is written to the output iterator which
226  * must be large enough to contain the results.
227  * The next available output iterator is returned.
228  */
229  template<typename InIter, typename OutIter,

233  OutIter& __pbegin) const
234  {
235  for (; __xbegin != __xend; ++__xbegin)
236  __pbegin++ = (*this)(__xbegin++);
237  return __pbegin;

240  // Could/should this be done by output iterator range?
241  template<size_type N>
242  constexpr void
243  eval(value_type __x, std::array<value_type, N>& __arr)
244  {
245  if (__arr.size() > 0)
246  {
247  __arr.fill(value_type{});
248  const size_type __sz = _M_coeff.size();
249  __arr[0] = this->coefficient(__sz - 1);
250  for (int __i = __sz - 2; __i >= 0; --__i)
251  {
252  int __nn = std::min(__arr.size() - 1, __sz - 1 - __i);
253  for (int __j = __nn; __j >= 1; --__j)
254  __arr[__j] = __arr[__j] * __x + __arr[__j - 1];
255  __arr[0] = __arr[0] * __x + this->coefficient(__i);
256  }
257  // Now put in the factorials.
258  value_type __fact = value_type(1);
259  for (size_t __i = 2; __i < __arr.size(); ++__i)
260  {
261  __fact *= value_type(__i);
262  __arr[__i] *= __fact;
263  }
264  }
265  }

267  /**
268  * Evaluate the polynomial and its derivatives at the point x.
269  * The values are placed in the output range starting with the
270  * polynomial value and continuing through higher derivatives.
271  */
272  template<typename OutIter>
273  constexpr void
274  eval(value_type __x, OutIter __b, OutIter __e)
275  {
276  if(__b != __e)
277  {
278  std::fill(__b, __e, value_type{});
279  constexpr size_type __sz = _M_coeff.size();
280  *__b = _M_coeff[__sz - 1];
281  for (int __i = __sz - 2; __i >= 0; --__i)
282  {
283  for (auto __it = std::reverse_iterator<OutIter>(__e);
284  __it != std::reverse_iterator<OutIter>(__b) - 1; ++__it)
285  *__it = *__it * __x + *(__it + 1);
286  *__b = *__b * __x + _M_coeff[__i];
287  }
288  // Now put in the factorials.
289  int __i = 0;
290  value_type __fact = value_type(++__i);
291  for (auto __it = __b + 1; __it != __e; ++__it)
292  {
293  __fact *= value_type(__i);
294  *__it *= __fact;
295  ++__i;
296  }
297  }
298  }

302  */
303  constexpr value_type
304  eval_even(value_type __x) const
305  {
306  if (this->degree() > 0)
307  {
308  auto __odd = this->degree() % 2;
309  value_type __poly(this->coefficient(this->degree() - __odd));

311  __poly = __poly * __x * __x + this->coefficient(__i);

313  }
314  else
315  return value_type{};

336  /**
337  * Evaluate the even part of the polynomial using a modification
338  * of Horner's rule which exploits the fact that the polynomial
339  * coefficients are all real.
340  *
341  * The algorithm is discussed in detail in:
342  * Knuth, D. E., The Art of Computer Programming: Seminumerical
343  * Algorithms (Vol. 2) Third Ed., Addison-Wesley, pp 486-488, 1998.
344  *
345  * If n is the degree of the polynomial,
346  * n - 3 multiplies and 4 * n - 6 additions are saved.
347  */
348  template<typename _Tp2>
349  constexpr auto
350  eval_even(std::complex<_Tp2> __z) const
351  -> decltype(value_type{} * std::complex<_Tp2>{})
352  {
353  if (this->degree() > 0)
354  {
355  const auto __zz = __z * __z;
356  const auto __r = _Tp{2} * std::real(__zz);
357  const auto __s = std::norm(__zz);
358  auto __odd = this->degree() % 2;
359  size_type __n = this->degree() - __odd;
360  auto __aa = this->coefficient(__n);
361  auto __bb = this->coefficient(__n - 2);
362  for (size_type __j = 4; __j <= __n; __j += 2)
363  __bb = this->coefficient(__n - __j)
364  - __s * std::exchange(__aa, __bb + __r * __aa);
365  return __aa * __zz + __bb;
366  }
367  else
368  return decltype(value_type{} * std::complex<_Tp2>{}){};
369  };

371  /**
372  * Evaluate the odd part of the polynomial using a modification
373  * of Horner's rule which exploits the fact that the polynomial
374  * coefficients are all real.
375  *
376  * The algorithm is discussed in detail in:
377  * Knuth, D. E., The Art of Computer Programming: Seminumerical
378  * Algorithms (Vol. 2) Third Ed., Addison-Wesley, pp 486-488, 1998.
379  *
380  * If n is the degree of the polynomial,
381  * n - 3 multiplies and 4 * n - 6 additions are saved.
382  */
383  template<typename _Tp2>
384  constexpr auto
385  eval_odd(std::complex<_Tp2> __z) const
386  -> decltype(value_type{} * std::complex<_Tp2>{})
387  {
388  if (this->degree() > 0)
389  {
390  const auto __zz = __z * __z;
391  const auto __r = _Tp{2} * std::real(__zz);
392  const auto __s = std::norm(__zz);
393  auto __even = (this->degree() % 2 == 0 ? 1 : 0);
394  size_type __n = this->degree() - __even;
395  auto __aa = this->coefficient(__n);
396  auto __bb = this->coefficient(__n - 2);
397  for (size_type __j = 4; __j <= __n; __j += 2)
398  __bb = this->coefficient(__n - __j)
399  - __s * std::exchange(__aa, __bb + __r * __aa);
400  return __z * (__aa * __zz + __bb);
401  }
402  else
403  return decltype(value_type{} * std::complex<_Tp2>{}){};
404  };
