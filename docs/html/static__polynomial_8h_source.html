143  /**
144  * Evaluate the polynomial at the input point.
145  */
146  constexpr value_type
147  operator()(value_type x) const
148  {
149  if (this->degree() > 0)
150  {
151  value_type poly(this->coefficient(this->degree()));
152  for (int i = this->degree() - 1; i >= 0; --i)
153  poly = poly * x + this->coefficient(i);
154  return poly;
155  }
156  else
157  return value_type{};
158  }

160  /**
161  * Evaluate the polynomial at the input point.
162  */
163  template<typename Tp2>
164  constexpr auto
165  operator()(Tp2 x) const
166  -> decltype(value_type{} * Tp2())
167  {
168  if (this->degree() > 0)
169  {
170  auto poly(this->coefficient(this->degree()) * Tp2(1));
171  for (int i = this->degree() - 1; i >= 0; --i)
172  poly = poly * x + this->coefficient(i);
173  return poly;
174  }
175  else
176  return value_type{};

179  /**
180  * The following polynomial evaluations are done using
181  * a modified of Horner's rule which exploits the fact that
182  * the polynomial coefficients are all real.
183  * The algorithm is discussed in detail in:
184  * Knuth, D. E., The Art of Computer Programming: Seminumerical
185  * Algorithms (Vol. 2) Third Ed., Addison-Wesley, pp 486-488, 1998.
186  *
187  * If n is the degree of the polynomial, n - 3 multiplies are
188  * saved and 4 * n - 6 additions are saved.
189  */
190  template<typename Tp2>
191  constexpr auto
192  operator()(std::complex<Tp2> z) const
193  -> decltype(value_type{} * std::complex<Tp2>{})
194  {
195  const auto r = Tp{2} * std::real(z);
196  const auto s = std::norm(z);
197  auto aa = this->coefficient(this->degree());
198  auto bb = this->coefficient(this->degree() - 1);
199  for (int j = 1; j <= this->degree(); ++j)
200  {
201  auto cc = s * aa;
202  aa = bb + r * aa;
203  bb = this->coefficient(this->degree() - j) - cc;
204  }
205  return aa * z + bb;
206  };

208  /**
209  * Evaluate the polynomial at a range of input points.
210  * The output is written to the output iterator which
211  * must be large enough to contain the results.
212  * The next available output iterator is returned.
213  */
214  template<typename InIter, typename OutIter,
215  typename = std::_RequireInputIter<InIter>>
216  constexpr OutIter
217  operator()(const InIter& xbegin, const InIter& xend,
218  OutIter& pbegin) const
219  {
220  for (; xbegin != xend; ++xbegin)
221  pbegin++ = (*this)(xbegin++);
222  return pbegin;
223  }

225  // Could/should this be done by output iterator range?
226  template<size_type N>
227  constexpr void
228  eval(value_type x, std::array<value_type, N>& arr)
229  {
230  if (arr.size() > 0)
231  {
232  arr.fill(value_type{});
233  const size_type sz = m_coeff.size();
234  arr[0] = this->coefficient(sz - 1);
235  for (int i = sz - 2; i >= 0; --i)
236  {
237  int nn = std::min(arr.size() - 1, sz - 1 - i);
238  for (int j = nn; j >= 1; --j)
239  arr[j] = arr[j] * x + arr[j - 1];
240  arr[0] = arr[0] * x + this->coefficient(i);
241  }
242  // Now put in the factorials.
243  value_type fact = value_type(1);
244  for (size_t i = 2; i < arr.size(); ++i)
245  {
246  fact *= value_type(i);
247  arr[i] *= fact;
248  }
249  }
250  }

252  /**
253  * Evaluate the polynomial and its derivatives at the point x.
254  * The values are placed in the output range starting with the
255  * polynomial value and continuing through higher derivatives.
256  */
257  template<typename OutIter>
258  constexpr void
259  eval(value_type x, OutIter b, OutIter e)
260  {
261  if(b != e)
262  {
263  std::fill(b, e, value_type{});
264  constexpr size_type sz = m_coeff.size();
265  *b = m_coeff[sz - 1];
266  for (int i = sz - 2; i >= 0; --i)
267  {
268  for (auto it = std::reverse_iterator<OutIter>(e);
269  it != std::reverse_iterator<OutIter>(b) - 1; ++it)
270  *it = *it * x + *(it + 1);
271  *b = *b * x + m_coeff[i];
272  }
273  // Now put in the factorials.
274  int i = 0;
275  value_type fact = value_type(++i);
276  for (auto it = b + 1; it != e; ++it)
277  {
278  fact *= value_type(i);
279  *it *= fact;
280  ++i;
281  }
282  }
283  }

285  /**
286  * Evaluate the even part of the polynomial at the input point.
287  */
288  constexpr value_type
289  eval_even(value_type x) const
290  {
291  if (this->degree() > 0)
292  {
293  auto odd = this->degree() % 2;
294  value_type poly(this->coefficient(this->degree() - odd));
295  for (int i = this->degree() - odd - 2; i >= 0; i -= 2)
296  poly = poly * x * x + this->coefficient(i);
297  return poly;
298  }
299  else
300  return value_type{};
301  }

304  * Evaluate the odd part of the polynomial at the input point.
305  */
306  constexpr value_type
307  eval_odd(value_type x) const
308  {
309  if (this->degree() > 0)

312  value_type poly(this->coefficient(this->degree() - even));
313  for (int i = this->degree() - even - 2; i >= 0; i -= 2)
314  poly = poly * x * x + this->coefficient(i);
315  return poly * x;
316  }
317  else

321  /**
322  * Evaluate the even part of the polynomial using a modification
323  * of Horner's rule which exploits the fact that the polynomial
324  * coefficients are all real.
325  *
326  * The algorithm is discussed in detail in:
327  * Knuth, D. E., The Art of Computer Programming: Seminumerical
328  * Algorithms (Vol. 2) Third Ed., Addison-Wesley, pp 486-488, 1998.
329  *
330  * If n is the degree of the polynomial,
331  * n - 3 multiplies and 4 * n - 6 additions are saved.
332  */
333  template<typename Tp2>
334  constexpr auto
335  eval_even(std::complex<Tp2> z) const
336  -> decltype(value_type{} * std::complex<Tp2>{})
337  {
338  if (this->degree() > 0)
339  {
340  const auto zz = z * z;
341  const auto r = Tp{2} * std::real(zz);
342  const auto s = std::norm(zz);
343  auto odd = this->degree() % 2;
344  size_type n = this->degree() - odd;
345  auto aa = this->coefficient(n);
346  auto bb = this->coefficient(n - 2);
347  for (size_type j = 4; j <= n; j += 2)
348  bb = this->coefficient(n - j)
349  - s * std::exchange(aa, bb + r * aa);
350  return aa * zz + bb;
351  }
352  else
353  return decltype(value_type{} * std::complex<Tp2>{}){};
354  };

356  /**
357  * Evaluate the odd part of the polynomial using a modification
358  * of Horner's rule which exploits the fact that the polynomial
359  * coefficients are all real.
360  *
361  * The algorithm is discussed in detail in:
362  * Knuth, D. E., The Art of Computer Programming: Seminumerical
363  * Algorithms (Vol. 2) Third Ed., Addison-Wesley, pp 486-488, 1998.
364  *
365  * If n is the degree of the polynomial,
366  * n - 3 multiplies and 4 * n - 6 additions are saved.
367  */
368  template<typename Tp2>
369  constexpr auto
370  eval_odd(std::complex<Tp2> z) const
371  -> decltype(value_type{} * std::complex<Tp2>{})
372  {
373  if (this->degree() > 0)
374  {
375  const auto zz = z * z;
376  const auto r = Tp{2} * std::real(zz);
377  const auto s = std::norm(zz);
378  auto even = (this->degree() % 2 == 0 ? 1 : 0);
379  size_type n = this->degree() - even;
380  auto aa = this->coefficient(n);
381  auto bb = this->coefficient(n - 2);
382  for (size_type j = 4; j <= n; j += 2)
383  bb = this->coefficient(n - j)
384  - s * std::exchange(aa, bb + r * aa);
385  return z * (aa * zz + bb);
386  }
387  else
388  return decltype(value_type{} * std::complex<Tp2>{}){};
389  };

391  /**
392  * Return the derivative of the polynomial.
393  */
394  constexpr StaticPolynomial<Tp, (Size > 1 ? Size - 1 : 1)>
395  derivative() const
396  {
397  StaticPolynomial<Tp, (Size > 1 ? Size - 1 : 1)> res;
398  for (size_type i = 1; i <= this->degree(); ++i)
399  res.coefficient(i - 1, i * m_coeff[i]);
400  return res;
401  }

403  /**
404  * Return the integral of the polynomial with given integration constant.
405  */
406  constexpr StaticPolynomial<Tp, Size + 1>
407  integral(value_type c = value_type{}) const
408  {
409  StaticPolynomial<Tp, Size + 1> res;
410  res.coefficient(0, c);
411  for (size_type i = 0; i <= this->degree(); ++i)
412  res.coefficient(i + 1, m_coeff[i] / value_type(i + 1));
413  return res;
414  }

713  /**
714  * Return the sum of two polynomials.
715  */
716  template<typename Tp, std::size_t SizeP, std::size_t SizeQ>
717  inline constexpr StaticPolynomial<Tp, std::max(SizeP, SizeQ)>
718  operator+(const StaticPolynomial<Tp, SizeP>& P,
719  const StaticPolynomial<Tp, SizeQ>& Q)
720  {
721  if constexpr (SizeP >= SizeQ)
722  {
723  StaticPolynomial<Tp, SizeP> R = P;
724  for (std::size_t i = 0; i < SizeQ; ++i)
725  R[i] += Q[i];
726  return R;
727  }
